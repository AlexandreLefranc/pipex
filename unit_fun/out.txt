#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>

void ft_putstr_fd(char *str, int fd)
{
	int i;

	i = 0;
	while (str[i] != '\0')
	{
		write(fd, &str[i], 1);
		i++;
	}
}

void redir_file_to_stdin(char *file)
{
	ssize_t	r;
	ssize_t fd;
	char	buffer[1024];

	fd = open(file, O_RDONLY);
	if (fd == -1)
	{
		write(2, "fd is -1", 9);
		return ;
	}
	r = read(fd, buffer, 1023);
	while (r != 0)
	{
		buffer[r] = '\0';
		ft_putstr_fd(buffer, 0);
		r = read(fd, buffer, 1023);
	}
	close(fd);
}

int main()
{
	char *filename = "redir_fd_to_stdin.c";

	int fdout = open("out.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
	dup2(fdout, STDIN_FILENO);
	redir_file_to_stdin(filename);
}
